git init[to intialize git]

git status [to list all  untracked files]

git add -A [to add all untracked files]

git commit -m "commit message" [to commit add file to local respository]

git clone [to take exact copy from remote respository to local workspace]

git pull [to take changes from remote respository to local workspace]

git push [to set changes from local respository to remote respository]
